git

代码管理工具，方便随时下载自己的代码

# 安装

官网 (opens new window)

安装完毕需要配置用户信息

git config --global user.name "Your Name"
git config --global user.email "email@example.com"
1
2

# 常用

# 初始化

在本地创建的文件夹执行

git init
1

# 关联

在本地创建仓库之后，需要在远程也创建仓库，然后本地仓库关联远程仓库

git remoto add origin <远程仓库的地址>
1

如果是第一次关联，需要进行ssh验证

# 提交

# 未跟踪的代码提交到缓存区
git add .  

# 本次提交的信息，方便记忆
git commit -m "初始化"

# 推送到远程仓库
git push -u origin master

git clone <远程仓库的地址>

# 修改推送的默认分支（错误）

# git config --global push.default <默认提交方式>
git branch --set-upstream-to=<远端库>/<远端分支> 修改的分支名

# 版本管理

# 查看修改

git status
1

# 查看记录

git log
1

# 回退版本

# 回退到上一个版本
git reset --hard HEAD^

# 跳转到指定版本
git reset --hard <版本号前几位>

# 分支管理

# 创建分支

git checkout -b <分支名>
1

# 查看分支

# 查看本地分支
git branch
# 查看本地分支关联（跟踪）的远程分支之间的对应关系，本地分支对应哪个远程分支
git branch -vv
1
2
3
4

# 关联分支

remotes代表远程

git branch --set-upstream-to remotes/<远程仓库名>/<分支名>
1

# 切换分支

git checkout <分支名>
1

# 合并分支

# 合并本地的分支
git merge <分支名>

# 合并远程的分支
git merge <远程仓库名>/<分支名>
1

# 删除分支

git branch -d <分支名>
1

# 修改分支名

git branch -m <原分支名> <新分支名>
