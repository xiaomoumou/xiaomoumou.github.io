# 2021.07.02

  • 关于存储部分
    • sessionStorage 只能存储字符串，不能存对象（数组，json 格式对象）
      • 通过 JSON.parse()、JSON.stringify()转换，这两个参数都接收转换函数
    • uni-app 的 uni.setStorageSync() 可以存对象
/**
 * 使用JSON.Stringify()转化Map数据时的处理方法
 * 解决JSON.Stringify()对Map数据不生效的问题
 * @param {*} key
 * @param {Map} value
 * @returns
 */
function replacer (key， value) {
  const originalObject = this[key]
  if (originalObject instanceof Map) {
    return {
      dataType: 'Map'，
      value: Array.from(originalObject.entries()) // or with spread: value: [...originalObject]
    }
  } else {
    return value
  }
}

/**
 * 使用JSON.parse()转换JSON.Stringify()序列化过的Map数据时使用
 * @param {*} key
 * @param {*} value
 * @returns
 */
function reviver (key， value) {
  if (typeof value === 'object' && value !== null) {
    if (value.dataType === 'Map') {
      return new Map(value.value)
    }
  }
  return value
}

const map = new Map()
map.set('name'，'张三')
map.set('age'，18)
console.log(map)

const mapToObj = JSON.stringify(map， replacer)
console.log(mapToObj) // {"dataType":"Map"，"value":[["name"，"张三"]，["age"，18]]}
const objToMap = JSON.parse(mapToObj，reviver)
console.log(objToMap) // Map(2) {'name' => '张三'， 'age' => 18}
