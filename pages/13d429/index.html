封装map组件


<template>
  <!-- components/BaiduMapView/index.vue -->
  <!-- 封装的baidu-map -->
  <baidu-map
    class="map-container"
    :center="center"
    :zoom="zoom"
    :scroll-wheel-zoom="true"
    @click="getPoint"
  >
    <!-- 地图的平移缩放控件 -->
    <bm-navigation anchor="BMAP_ANCHOR_TOP_RIGHT" />
    <bm-marker
      :position="center"
      :dragging="true"
      animation="BMAP_ANIMATION_BOUNCE"
      :zoom="zoom"
      :raise-on-drag="true"
      @dragend="getPoint"
    >
      <bm-label
        content=""
        :label-style="{ color: 'red', fontSize: '24px' }"
        :offset="{ width: -35, height: 30 }"
      />
    </bm-marker>

    <!-- 位置检索、周边检索和范围检索 -->
    <bm-local-search
      :keyword="keyword"
      :panel="false"
      :location="location"
      :auto-viewport="autoViewport"
      @searchcomplete="handleSearchChange"
    />
    <!-- 定位控件 -->
    <bm-geolocation
      anchor="BMAP_ANCHOR_BOTTOM_RIGHT"
      :show-address-bar="true"
      :auto-location="true"
    />
    <!-- 行政区域的边界 -->
    <bm-boundary :name="center" :stroke-weight="2" stroke-color="blue" />
  </baidu-map>
</template>

<script>
export default {
  name: 'BaiduMapView',
  props: {
    // 所有省、市、区、镇信息
    cityData: {
      type: Array,
      default: () => {
        return []
      }
    },
    // 地址省、市、区、镇的id数组
    address: {
      type: Array,
      default: () => {
        return []
      }
    },
    addressInfo: {
      type: Object,
      default: () => {
        return {}
      }
    },
    // 搜索的关键字
    mapKeyword: {
      type: String,
      default: () => {
        return ''
      }
    },
    // 搜索的区域
    location: {
      type: String,
      default: ''
    }
  },
  data () {
    return {
      center: {
        lng: 113.271429,
        lat: 23.135336
      },
      zoom: 15,
      keyword: '',
      autoViewport: true
    }
  },
  watch: {
    address: {
      handler (val) {
        this.areaChange(val)
      },
      deep: true
    },
    addressInfo: {
      handler (val) {
        this.center = {}
        this.center.lng = val.lon
        this.center.lat = val.lat
      },
      deep: true
    }
  },
  methods: {
    // 选择区域变化时触发
    areaChange (addressAll) {
      // 获取最后一级的街道名
      // const areaName = this.$trans.readTreeDataById(
      //   this.cityData,
      //   addressAll
      // ).areaName
      // 获取省市区街四级名字
      let areaName = ''
      this.cityData.forEach((province) => {
        if (province.areaId === addressAll[0]) {
          areaName += province.areaName
          province.children.forEach((city) => {
            if (city.areaId === addressAll[1]) {
              areaName += city.areaName
              city.children.forEach((district) => {
                if (district.areaId === addressAll[2]) {
                  areaName += district.areaName
                }
              })
            }
          })
        }
      })
      let zoom = 10
      // 根据不同的详细度，显示不同的缩放级别
      switch (addressAll.length) {
        case 1:
          zoom = 8
          break
        case 2:
          zoom = 11
          break
        case 3:
          zoom = 12
          break
        case 4:
          zoom = 16
          break
      }
      const mapConfig = {
        areaName,
        zoom
      }
      this.setCenterAndZoom(mapConfig)
    },
    handleSearchChange (LocalResult) {
      this.$emit('searchChange', LocalResult)
    },
    search () {
      this.keyword = this.mapKeyword
      this.zoom = 14
    },
    // 设置中心和缩放比例
    setCenterAndZoom (e) {
      if (e.target) {
        const { lng, lat } = e.target.getCenter()
        this.center.lng = lng
        this.center.lat = lat
        this.zoom = e.target.getZoom()
      } else if (e.search) {
        const { lng, lat } = e
        this.center = {}
        this.center.lng = lng
        this.center.lat = lat
      } else {
        const { areaName, zoom } = e
        this.center = areaName
        this.zoom = zoom
      }
    },
    // 平移和单击时渲染一个红色的坐标
    getPoint (e) {
      this.center = {}
      this.center.lng = e.point.lng
      this.center.lat = e.point.lat
      var geoc = new BMap.Geocoder()
      geoc.getLocation(e.point, (rs) => {
        console.log(rs)
        this.$emit('pointChange', rs)
      })
    }
  }
}
</script>

<template>
  <div id="contact">
    <el-cascader
      v-model="addAddressData.addressAll"
      style="width: 400px"
      :options="cityData"
      :props="{
        label: 'areaName',
        value: 'areaId'
      }"
      @change="getCityData"
      @expand-change="handleAreaChange"
    />

<script>
import BaiduMapComponent from '@/components/BaiduMapView'

export default {
  components: { BaiduMapComponent },
  data(){
    // 整个地图的“省市区街”数据，略缩
    cityData:[
        [{
            "areaId": 110000000000,
            "areaName": "北京市",
            "parentId": 0,
            "children": [{
                "areaId": 110100000000,
                "areaName": "市辖区",
                "parentId": 110000000000,
                "children": [{
                    "areaId": 110101000000,
                    "areaName": "东城区",
                    "parentId": 110100000000,
                    "children": [{
                        "areaId": 110101001000,
                        "areaName": "东华门街道",
                        "parentId": 110101000000
                    }, {
                        "areaId": 110101002000,
                        "areaName": "景山街道",
                        "parentId": 110101000000
                    }, {
                        "areaId": 110101003000,
                        "areaName": "交道口街道",
                        "parentId": 110101000000
                    }]
                }]
            }]
        }]
    ]
  },
  methods:{
    // 当<el-cascader>组件更改时，修改最后一级之后触发
    getCityData (e) {
      this.addressIds = e
      this.addAddressData.cityData = e
      this.city = this.$trans.readTreeDataById(this.cityData, e).areaName
    },
    // 当<el-cascader>组件更改之后，每次修改都触发
     handleAreaChange (val) {
      this.addressIds = val
    },
  }
}
</script>

工具函数

// 获取树形结构前几层的数据
const setTreeHierarchyIndex = (data, maxLevel, currLevel) => {
  if (!(data instanceof Array)) {
    throw new TypeError('The data should be an array!')
  }
  const result = []
  for (let k = 0; k < data.length; k++) {
    const temp = data[k]
    const newNode = {
      ...temp,
      hierarchyIndex: currLevel + 1,
      children: null
    }
    delete newNode.children
    // 是否在指定层级范围
    if (currLevel >= maxLevel) continue
    if (temp.children && temp.children.length > 0) {
      currLevel++
      newNode.children = setTreeHierarchyIndex(
        temp.children,
        maxLevel,
        currLevel
      )
      if (!newNode.children.length) {
        delete newNode.children
      }
      currLevel--
    }
    result.push(newNode)
  }
  return result
}
// 根据id值获取树状数据的对象
 readTreeDataById (cityData, e, toggle) {
    let result = {}
    e.forEach((element, index) => {
      if (index === 0) {
        result = this.recursionReadData(cityData, element)
      } else {
        if (toggle) {
          console.log('result.data')
          console.log(result.data)
        }

        result = this.recursionReadData(result.data, element)
      }
    })
    return result
  }
    // 递归查找树形结构
  recursionReadData (cityData, id) {
    let areaName = ''
    let data = {}
    let areas = {}
    for (let i = 0; i < cityData.length; i++) {
      // eslint-disable-next-line eqeqeq
      if (cityData[i].areaId === id) {
        // children若为空数组，则将children设为undefined
        areaName = cityData[i].areaName
        data = cityData[i].children
        areas = cityData[i]
      }
    }
    return {
      areaName,
      data,
      areas
    }
  }
