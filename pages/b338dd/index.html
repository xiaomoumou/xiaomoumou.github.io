# 格式化时间

/**
 * 格式化时间
 * Parse the time to string
 * @param {(Object|string|number)} time
 * @param {string} cFormat
 * @returns {string | null}
 */
export function parseTime(time, cFormat) {
  if (arguments.length === 0) {
    return null
  }
  const format = cFormat || '{y}-{m}-{d} {h}:{i}:{s}'
  let date
  if (typeof time === 'object') {
    date = time
  } else {
    if (typeof time === 'string' && /^[0-9]+$/.test(time)) {
      time = parseInt(time)
    }
    if (typeof time === 'number' && time.toString().length === 10) {
      time = time * 1000
    }
    date = new Date(time)
  }
  const formatObj = {
    y: date.getFullYear(),
    m: date.getMonth() + 1,
    d: date.getDate(),
    h: date.getHours(),
    i: date.getMinutes(),
    s: date.getSeconds(),
    a: date.getDay()
  }
  const time_str = format.replace(/{([ymdhisa])+}/g, (result, key) => {
    const value = formatObj[key]
    // Note: getDay() returns 0 on Sunday
    if (key === 'a') {
      return ['日', '一', '二', '三', '四', '五', '六'][value]
    }
    return value.toString().padStart(2, '0')
  })
  return time_str
}

/**
 * 固定的时间戳 转换为 显示多少分钟前
 * @param {number} time
 * @param {string} option
 * @returns {string}
 */
export function formatTime(time, option) {
  if (('' + time).length === 10) {
    time = parseInt(time) * 1000
  } else {
    time = +time
  }
  const d = new Date(time)
  const now = Date.now()

  const diff = (now - d) / 1000

  if (diff < 30) {
    return '刚刚'
  } else if (diff < 3600) {
    // less 1 hour
    return Math.ceil(diff / 60) + '分钟前'
  } else if (diff < 3600 * 24) {
    return Math.ceil(diff / 3600) + '小时前'
  } else if (diff < 3600 * 24 * 2) {
    return '1天前'
  }
  if (option) {
    return parseTime(time, option)
  } else {
    return (
      d.getMonth() +
      1 +
      '月' +
      d.getDate() +
      '日' +
      d.getHours() +
      '时' +
      d.getMinutes() +
      '分'
    )
  }
}

# 深拷贝

/**
 * 深拷贝
 * @param {*} source
 * @returns
 */
export function deepClone(source) {
  if (!source && typeof source !== 'object') {
    throw new Error('error arguments', 'deepClone')
  }
  const targetObj = source.constructor === Array ? [] : {}
  Object.keys(source).forEach(keys => {
    if (source[keys] && typeof source[keys] === 'object') {
      targetObj[keys] = deepClone(source[keys])
    } else {
      targetObj[keys] = source[keys]
    }
  })
  return targetObj
}

# 深合并

以target为主 数组直接添加合并 对象相同属性名合并，不同添加

function merge(target, source) {
  //类型不一致返回target
  if (typeof target !== typeof source || (!target instanceof Array && !source instanceof Array)) {
    return target;
  }
  if (target instanceof Array) {
    target = target.concat(source);
  } else if (typeof target === ‘object’) {
    for (let key in target) {
      if (source[key]) {
        target[key] = merge(target[key], source[key]);
      }
    }
    for (let key in source) {
      if (!target[key]) {
        target[key] = source[key];
      }
    }
  } else {
  //除了object array类型其他都返回target
  }
  return target;
}

// 不同属性合并
let a={b:‘1111’}
let b={c:‘222’}
a=merge(a,b) // {b:‘1111’,c:‘222’}

// 相同属性覆盖
let a={b:‘1111’,d:‘333’}
let b={c:‘222’,d:‘444’}
a=merge(a,b) // {b: “1111”, d: “333”, c: “222”}

// 相同属性对象类型合并
let a={b:‘1111’,d:{e:‘555’}}
let b={c:‘222’,d:{f:‘666’}}
a=merge(a,b) // {b: “1111”, d: {e: “555”, f: “666”}, c: “222”}

/**
 * 数组去重，获得后面出现的数据
 * @param {Array} arr 
 * @param {String} sign 
 */
export function duplicateRemoval(arr,sign){
  //把数组转对象
  var obj = {}; 
  var j ;
  for(var i=0 ,j= arr.length; i<j;i++){
      obj[arr[i][sign]] = arr[i] ;
  }
  // 对象转数组
  var arr = [];//私有对象
  for(var attr in obj){
      if(obj.hasOwnProperty(attr)){        //YUI底层代码
          arr.push(obj[attr]);
      }
  }
  return  arr;
}

# 获取树结构的一部分

/**
 * 当对象里的某个key对应是value时，返回这个key所在的对象
 * @param {Object} data
 * @param {String} field  // 特定的参数，比如'id'
 * @param {*} value
 * @returns
 */
export function findObject(data, field, value) {
  let result = null
  for (const key in data) {
    if (key === field && data[key] === value) {
      result = data
    }
    if (typeof (data[key]) === 'object') {
      result = findObject(data[key], field, value)
    }
    if (result) {
      return result
    }
  }
  return null
}
